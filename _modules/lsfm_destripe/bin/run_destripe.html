def bool_args(arg):
    if ("false" == arg) or ("False" == arg):
        return False
    elif ("true" == arg) or ("True" == arg):
        return True

class Args(argparse.Namespace):

    DEFAULT_FIRST = 10
    DEFAULT_SECOND = 20

    def __init__(self):
        # Arguments that could be passed in through the command line
        self.first = self.DEFAULT_FIRST
        self.second = self.DEFAULT_SECOND
        self.debug = False
        #
        self.__parse()

def main():
    try:
        args = Args()
        dbg = args.debug

        exe = DeStripe(
            args.loss_eps,
            args.qr,
            args.resample_ratio,
            args.GF_kernel_size_train,
            args.GF_kernel_size_inference,
            args.hessian_kernel_sigma,
            args.sampling_in_MSEloss,
            args.isotropic_hessian,
            args.lambda_tv,
            args.lambda_hessian,
            args.inc,
            args.n_epochs,
            args.wedge_degree,
            args.n_neighbors,
            args.fast_GF,
            args.require_global_correction,
            args.fusion_GF_kernel_size,
            args.fusion_Gaussian_kernel_size,
        )
        out = exe.train(
            args.X1_path,
            args.is_vertical,
            args.angle_offset,
            args.X2_path,
            args.mask_path,
            args.boundary,
        )
        if not isinstance(out, tuple):
            OmeTiffWriter.save(out, args.save_path, dim_order="ZYX")
        else:
            OmeTiffWriter.save(
                out[0],
                args.save_path,
                dim_order="ZYX",
            )
            OmeTiffWriter.save(
                out[1],
                args.save_path_top_or_left_view,
                dim_order="ZYX",
            )
            OmeTiffWriter.save(
                out[2],
                args.save_path_bottom_or_right_view,
                dim_order="ZYX",
            )

    except Exception as e:
        log.error("=============================================")
        if dbg:
            log.error("\n\n" + traceback.format_exc())
            log.error("=============================================")
        log.error("\n\n" + str(e) + "\n")
        log.error("=============================================")
        sys.exit(1)
